# Preview-Mode Fix Agent Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When the operator triggers a fix run, land the changes on an isolated `back-office/preview/<job-id>` branch in the target repo (not on `main`) and emit a machine-readable `preview.json` describing what changed, so the Review & Approve UI (Plan 4) can walk the operator through a checklist.

**Architecture:** Add a `--preview` flag to `agents/fix-bugs.sh` that (1) creates a preview branch before the fix agent runs, (2) passes the branch name into the agent prompt so every commit lands there, (3) after the agent returns, snapshots the diff into `results/<repo>/preview-<job-id>.json`, (4) pushes the branch to `origin` so a reviewer can open a GitHub compare URL. The preview JSON is generated by a small Python module (`backoffice/preview.py`) that is trivially unit-testable — the shell script just calls it.

**Tech Stack:** Bash, Python stdlib (`subprocess`, `json`, `datetime`), git.

```python
"""Preview artifact generator for the fix agent.

The fix-bugs.sh script calls :func:`build_preview` after the agent has
finished committing on the preview branch.  The returned dict is written
to ``results/<repo>/preview-<job-id>.json`` and consumed by the Review
& Approve panel in the dashboard.

The module deliberately has no shell-specific code so the shape can be
unit tested against a real tmp git repo.
"""
from __future__ import annotations

import re
import subprocess
from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path
from typing import Any


# ── Inputs / outputs ────────────────────────────────────────────────────────

@dataclass
class PreviewInputs:
    repo_path: Path
    repo_name: str
    job_id: str
    branch: str
    base_ref: str
    findings_addressed: list[dict[str, Any]] = field(default_factory=list)
    remote_url: str | None = None


# ── git helpers ─────────────────────────────────────────────────────────────

def _git(repo_path: Path, *args: str) -> str:
    result = subprocess.run(
        ["git", *args],
        cwd=str(repo_path),
        capture_output=True,
        text=True,
        check=True,
    )
    return result.stdout.strip()


def _rev_parse(repo_path: Path, ref: str) -> str:
    return _git(repo_path, "rev-parse", ref)


def _numstat(repo_path: Path, base: str, head: str) -> list[dict[str, Any]]:
    """Per-file insertion/deletion counts between base and head."""
    raw = _git(repo_path, "diff", "--numstat", f"{base}..{head}")
    out: list[dict[str, Any]] = []
    for line in raw.splitlines():
        ins, dels, path = (line.split("\t") + ["", ""])[:3]
        try:
            ins_n = int(ins)
            dels_n = int(dels)
        except ValueError:
            # Binary files report "-\t-\tpath" — skip the counts.
            ins_n = 0
            dels_n = 0
        out.append({"file": path, "insertions": ins_n, "deletions": dels_n})
    return out


def _commits(repo_path: Path, base: str, head: str) -> list[dict[str, str]]:
    raw = _git(repo_path, "log", "--format=%H%x1f%s", f"{base}..{head}")
    out: list[dict[str, str]] = []
    for line in raw.splitlines():
        if "\x1f" not in line:
            continue
        sha, subject = line.split("\x1f", 1)
        out.append({"sha": sha, "subject": subject})
    return out


# ── GitHub URL derivation ──────────────────────────────────────────────────

_GITHUB_SSH = re.compile(r"^git@github\.com:(?P<owner>[^/]+)/(?P<repo>.+?)(\.git)?$")
_GITHUB_HTTPS = re.compile(r"^https://github\.com/(?P<owner>[^/]+)/(?P<repo>.+?)(\.git)?$")


def _compare_url(remote_url: str | None, base_ref: str, head_branch: str) -> str | None:
    if not remote_url:
        return None
    for pat in (_GITHUB_SSH, _GITHUB_HTTPS):
        m = pat.match(remote_url)
        if m:
            owner, repo = m["owner"], m["repo"]
            return f"https://github.com/{owner}/{repo}/compare/{base_ref}...{head_branch}"
    return None


# ── Checklist builder ──────────────────────────────────────────────────────

def _checklist(findings: list[dict[str, Any]]) -> list[dict[str, Any]]:
    """One checklist item per addressed finding."""
    items: list[dict[str, Any]] = []
    for f in findings:
        fid = str(f.get("id", "")) or "UNKNOWN"
        items.append({
            "finding_id": fid,
            "severity": f.get("severity", "info"),
            "trust_class": f.get("trust_class", "objective"),
            "title": f.get("title", ""),
            "file": f.get("file", ""),
            "line": f.get("line"),
            "verify": (
                "Objective check: confirm the fix compiles, tests pass, "
                "and the referenced file:line shows the change."
                if f.get("trust_class") == "objective"
                else "Advisory check: this fix is a judgement call — read "
                     "the diff and confirm it matches intent before approving."
            ),
        })
    return items


# ── Entry point ────────────────────────────────────────────────────────────

def build_preview(inputs: PreviewInputs) -> dict[str, Any]:
    base_sha = _rev_parse(inputs.repo_path, inputs.base_ref)
    head_sha = _rev_parse(inputs.repo_path, "HEAD")

    return {
        "version": 1,
        "job_id": inputs.job_id,
        "repo": inputs.repo_name,
        "branch": inputs.branch,
        "base_ref": inputs.base_ref,
        "base_sha": base_sha,
        "head_sha": head_sha,
        "compare_url": _compare_url(inputs.remote_url, inputs.base_ref, inputs.branch),
        "changes": _numstat(inputs.repo_path, base_sha, head_sha),
        "commits": _commits(inputs.repo_path, base_sha, head_sha),
        "checklist": _checklist(inputs.findings_addressed),
        "created_at": datetime.now(timezone.utc).isoformat(),
    }
```

```python
def _derive_remote(repo_path):
    import subprocess
    try:
        return subprocess.check_output(
            ["git", "remote", "get-url", "origin"],
            cwd=str(repo_path), text=True,
        ).strip()
    except subprocess.CalledProcessError:
```

### Task 3: `--preview` flag on `agents/fix-bugs.sh`

**Files:**
- Modify: `agents/fix-bugs.sh`

The current script (reviewed at the plan-authoring time) has this shape:
- Parses `$1` as target repo and `--sync` / `--deploy` flags
- Reads findings, echoes a banner, builds a prompt, launches the agent runner
- Optionally runs deploy/sync afterwards

We are adding:
- A new `--preview` flag.
- Before invoking the agent: `git checkout -b back-office/preview/<job-id>` in the target repo.
- After the agent returns: `git push --set-upstream origin <branch>`, then call `python -m backoffice preview` to write `results/<repo>/preview-<job-id>.json`.
- The fix-prompt gains a line: "Commit on the current branch (already checked out by the launcher); do not merge to main."

```python
def iter_preview_files(results_root):
    """Yield every preview-*.json under results_root.

    Used by :mod:`backoffice.sync.engine` to upload operator review
    artifacts alongside the findings.
    """
    from pathlib import Path
    root = Path(results_root)
    if not root.exists():
        return
    for p in sorted(root.glob("*/preview-*.json")):
        yield p
```

## Remember

- Preview branches use a stable prefix (`back-office/preview/`) so target repos can ignore-list them from their deploy pipelines if they want to.
- The base ref is snapshotted at checkout time — if the agent runs long and `main` moves, the compare URL still points at the right base.
- Every preview generates one artifact per job. Don't reuse `preview.json` with a variable payload — reviewers need a permanent handle, and keeping each run addressable by job-id avoids race conditions with the Review panel's polling.
- `.preview-base` is written under `results/<repo>/` intentionally — if you write it inside the target repo, it pollutes their working tree.
